  Create a particle system with wall logic! 

            

                // declare object array

                Particle[] particles = new Particle[100];

                void setup()
                {

                // define canvas size, background colour and blend mode 

                  size(500, 500);
                  background(255);
                  blendMode(MULTIPLY);

                // initialise object array using for loop
                
                  for (int i = 0; i < particles.length; i++)
                  {
                    particles[i] = new Particle();
                  }
                }

                // display objects, updated them and apply logic using for loop
                
                void draw()
                {
                  for (int i = 0; i < particles.length; i++)
                  {
                    particles[i].display();
                    particles[i].update();
                    particles[i].logic();
                  }

                // save unique frames to folder (maybe mouseClicked is better,
                // or use a separate function, can't remember, just play around
                // with it if you get too many frames from this)
                  
                  if(mousePressed)
                  {
                  saveFrame("particle_system01-######.jpg");
                  }
                  
                }

                // make particle class with random x + y positions
                // use a fixed radius or a random one
                // give particles some random directions
                
                class Particle {
                  float x = random(width);
                  float y = random(height);
                
                  //float radius = 3;
                  float radius = random(0.2, 125);
                
                  float xdirection = random(-1, 1);
                  float ydirection = random(-1, 1);
                  
                    Particle()
                  {
                    
                  }

                  // give the particle some shape and colour you like
                
                  void display()
                  {
                    noStroke();
                    fill(0, 50, 120, 5);
                    ellipse(x, y, radius, radius);
                  }

                  // update the direction
                
                  void update()
                  {
                    x += xdirection;
                    y += ydirection;
                  }

                  // if the particles x / y position reaches an edge reverse said direction so it bounces
                
                  void logic()
                  {
                    if (x > width)
                    {
                      xdirection = -xdirection;
                    }
                    if (x < 0)
                    {
                      xdirection = -xdirection;
                    }
                    if (y > height)
                    {
                      ydirection = -ydirection;
                    }
                    if (y < 0)
                    {
                      ydirection = -ydirection;
                    }
                  }
                }
